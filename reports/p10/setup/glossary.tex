\newglossaryentry{symmetric}
{name={symmetric}, description={$a_{i,j} = a_{j,i}$ for $i, j = 0, \dots , N -
    1$~\cite{Press:2007:NRE:1403886}}}

\newglossaryentry{baudrate}
{name={baud rate}, description={The baud rate is the rate at which information can be transferred as a
    wireless signal~\cite{website:baudrate-mathworks}, and is equivalent to bits per second}}

\newglossaryentry{lt-matrix}
{name={lower triangular matrix}, description={A square matrix where all entries above the diagonal are
    zero~\cite{Press:2007:NRE:1403886}}}

\newglossaryentry{conjugate-transpose}
{name={conjugate transpose}, description={The element in the $i$-th row and $j$-th column is equal to the
    complex conjugate of the element in the $j$-th row and $i$-th column for all indices $i$ and
    $j$~\cite{Press:2007:NRE:1403886}}}

\newglossaryentry{pd-matrix}
{name={positive-definite matrix}, description={A matrix is positive-definite if $\forall n \in M$, such that
    $eigenvalue(n) > 0$~\cite{website:positive-definite-wolfram}}}

\newglossaryentry{hop}
{name={hop}, description={The amount of nodes between source and destination}}

\newglossaryentry{octet}
{name={octet}, description={A unit of digital information consisting of exactly eight bits}}

\newglossaryentry{lowerlayer}
{name={lower layer}, description={The software simulating radio hardware}}

\newglossaryentry{upperlayer}
{name={upper layer}, description={The software implementation of a \acrshort{manet} protocol, e.g., an
    implementation of the \acrshort{lmac} protocol}}

\newglossaryentry{pathloss}
{name={path loss}, description={The signal loss inflicted by the propagation of a electromagnetic wave from
    transmitter to receiver~\cite[p.~10]{paper:linkmodel}}}
